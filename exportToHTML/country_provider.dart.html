18     Future<void> getAllCountries() async {

32       List<Country> results = [];
33       // List<Country> results2 = [];
34       if (_searchString.isEmpty) {
35         // if the search field is empty or only contains white-space, we'll display all users

38       } else {
39         // results = _countries;
40         results = _filterCountries.where((c) {

42           final code = c.callingCodes.first;
43           final searchLower = _searchString.toLowerCase();
44           //
45           // display list of countries in which entered number or keyword present
46           return countryLower.contains(searchLower) || code.contains(searchLower);
47         }).toList();
48       }
49       print(results);
50       return results;

54     }
