115    Widget buildCountry(Country country) {
116      return ListTile(
117        leading: Text(
118          '+${country.dialCode}',
119          style: const TextStyle(
120            color: Color(0xFF594C74),
121            fontWeight: FontWeight.w500,
122            fontSize: 17,
123          ),
124        ),
125        title: Text(
126          country.name,
127          style: const TextStyle(
128            color: Color(0xFFFFFFFF),
129            fontWeight: FontWeight.w500,
130            fontSize: 17,
131          ),
132        ),
133        dense: true,
134        visualDensity: const VisualDensity(vertical: -1.5), // to compact
135        onTap: () {
136          _selectItem('+${country.dialCode}');
137        },
138      );
139    }
140  
141    Widget buildSearch() => SearchWidget(
142          text: query,
143          hintText: 'Counry name or counry code',
144          onChanged: searchCountry,
145        );
146  
147    void searchCountry(String query) {
148      final countries = allCountries.where((country) {
149        final nameLower = country.name.toLowerCase();
150        final searchLower = query.toLowerCase();
151  
152        return nameLower.contains(searchLower);
153      }).toList();
154  
155      setState(() {
156        this.query = query;
157        this.countries = countries;
158      });
159    }
